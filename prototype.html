Pascal IF Statement Examples

Boolean Example
var
  flag: Boolean;
begin
  flag := True;
  if flag then
    ShowMessage('Boolean is True')
  else
    ShowMessage('Boolean is False');
Integer Example
var
  age: Integer;
begin
  age := 30;
  if age > 18 then
    ShowMessage('Adult')
  else
    ShowMessage('Minor');
Double (Float/Real) Example
var
  price: Double;
begin
  price := 99.99;
  if price < 100 then
    ShowMessage('Cheap')
  else
    ShowMessage('Expensive');
Char Example
var
  letter: Char;
begin
  letter := 'A';
  if letter = 'A' then
    ShowMessage('First letter of alphabet');
String Example
var
  username: String;
begin
  username := 'admin';
  if username = 'admin' then
    ShowMessage('Welcome, admin!')
  else
    ShowMessage('Access denied');
Enumeration Example
type
  TStatus = (stOnline, stOffline, stAway);
var
  status: TStatus;
begin
  status := stOnline;
  if status = stOnline then
    ShowMessage('User is online');
Set Example
type
  TDay = (Mon, Tue, Wed, Thu, Fri, Sat, Sun);
  TDays = set of TDay;
var
  workDays: TDays;
begin
  workDays := [Mon, Tue, Wed, Thu, Fri];
  if Sat in workDays then
    ShowMessage('Saturday is a work day')
  else
    ShowMessage('Saturday is a day off');
Pointer Example
var
  ptr: Pointer;
begin
  ptr := nil;
  if ptr = nil then
    ShowMessage('Pointer is nil');
Class/Object Example
var
  frm: TForm;
begin
  frm := TForm.Create(nil);
  if frm <> nil then
    ShowMessage('Form created successfully');
  frm.Free;
Variant Example
var
  data: Variant;
begin
  data := 'Test';
  if VarType(data) = varString then
    ShowMessage('It''s a string');
Record Example
type
  TPerson = record
    Name: String;
    Age: Integer;
  end;
var
  p: TPerson;
begin
  p.Name := 'Alice';
  p.Age := 25;
  if p.Age > 20 then
    ShowMessage(p.Name + ' is over 20');
Array Example
var
  numbers: array[1..3] of Integer;
begin
  numbers[1] := 10;
  if numbers[1] = 10 then
    ShowMessage('First number is 10');
Dynamic Array Example
var
  dynArr: array of Integer;
begin
  SetLength(dynArr, 2);
  dynArr[0] := 5;
  if dynArr[0] = 5 then
    ShowMessage('Dynamic array[0] = 5');
File Existence Example
if FileExists('data.txt') then
  ShowMessage('File exists')
else
  ShowMessage('File not found');
Interface Example
type
  IMyInterface = interface
    procedure DoSomething;
  end;
  TMyObject = class(TInterfacedObject, IMyInterface)
    procedure DoSomething;
  end;
procedure TMyObject.DoSomething;
begin
  ShowMessage('Doing something');
end;
var
  intf: IMyInterface;
begin
  intf := TMyObject.Create;
  if Assigned(intf) then
    intf.DoSomething;
Multiple Conditions Example
var
  x, y: Integer;
begin
  x := 5;
  y := 10;
  if (x < y) and (y < 20) then
    ShowMessage('x < y < 20');
